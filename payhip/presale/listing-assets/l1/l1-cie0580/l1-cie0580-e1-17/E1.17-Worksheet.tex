\SectionHeader{Section A: Exponential Change Fundamentals}{Growth and decay multipliers over fixed periods}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A population of \(1200\) grows by \(5\%\) in one year. Find the new population.\MarkOne \AnswerBox
  \item A value of \(600\) decays by \(12\%\) in one year. Find the new value.\MarkOne \AnswerBox
  \item Find the value of \(5000(1.08)^3\).\MarkOne \AnswerBox
  \item A machine worth \(900\) depreciates by \(15\%\) per year for 2 years. Find its value after 2 years.\MarkOne \AnswerBox
  \item Write a formula for the value \(V\) after \(n\) years if initial value is \(P\) and annual rate is \(r\%\) growth.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Multi-Period Exponential Models}{Repeated percentage change and half-life reasoning}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Calculate the final amount when \pounds2000 is invested at \(4.5\%\) compound interest for 3 years.\MarkOne \AnswerBox
  \item A car worth \pounds18000 depreciates by \(18\%\) each year for 4 years. Find its value after 4 years.\MarkOne \AnswerBox
  \item A culture doubles every 6 hours. It starts at \(3.2\times10^5\). Find the number after 24 hours.\MarkOne \AnswerBox
  \item A substance has half-life 8 hours. If initial mass is 960 g, find the mass after 24 hours.\MarkOne \AnswerBox
  \item Solve \(500(1.03)^t=650\) for \(t\), correct to 2 decimal places.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Exponential Equations}{Solving for time and interpreting model constraints}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item If \(N=1500(1.12)^n\), find the smallest integer \(n\) such that \(N>3000\).\MarkOne \AnswerBox
  \item If \(M=7000(0.94)^n\), find the smallest integer \(n\) such that \(M<4000\).\MarkOne \AnswerBox
  \item A town population is \(25000\) and grows by \(2.4\%\) per year for 5 years. Find the increase in population over the 5 years.\MarkOne \AnswerBox
  \item A radioactive sample has mass 40 g after 15 hours. Half-life is 5 hours. Find the initial mass.\MarkOne \AnswerBox
  \item A quantity follows \(V=V_0\times2^{t/9}\). Find \(\frac{V}{V_0}\) when \(t=27\).\MarkOne \AnswerBox
\end{enumerate}
