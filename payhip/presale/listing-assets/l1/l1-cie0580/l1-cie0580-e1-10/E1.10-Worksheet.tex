\SectionHeader{Section A: Reading Accuracy Statements}{Error intervals, upper bounds, and lower bounds}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A length \(L\) is recorded as \(8.4\) cm correct to 1 decimal place. Write the error interval for \(L\).\MarkOne \AnswerBox
  \item A mass is \(2.73\) kg correct to 2 decimal places. Find the maximum absolute error.\MarkOne \AnswerBox
  \item A value is \(560\) correct to 2 significant figures. Find the lower bound.\MarkOne \AnswerBox
  \item A radius is \(12\) cm correct to the nearest cm. Find the upper bound.\MarkOne \AnswerBox
  \item A time is \(37.2\) s correct to 1 decimal place. Find the greatest possible true time.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Bounds in Calculations}{Using limits of accuracy in area and speed contexts}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A rectangle has length \(6.8\) cm and width \(4.3\) cm, each correct to 1 decimal place. Find the upper bound for its area.\MarkOne \AnswerBox
  \item Using Q6, find the lower bound for the area.\MarkOne \AnswerBox
  \item A car travels \(154\) km correct to the nearest km in \(2.6\) h correct to 1 decimal place. Find the upper bound for its speed.\MarkOne \AnswerBox
  \item Using Q8, find the lower bound for the speed.\MarkOne \AnswerBox
  \item A voltage is \(240\) V correct to the nearest 10 V. Find the upper bound.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Absolute and Relative Error}{Percentage error reasoning in exam-style applications}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item The true value is \(49.8\) and the measured value is \(50.1\). Find the absolute error.\MarkOne \AnswerBox
  \item Using Q11, find the percentage error, correct to 2 decimal places.\MarkOne \AnswerBox
  \item A scale reading is \(7.35\) kg correct to the nearest \(0.01\) kg. Find the maximum percentage error, correct to 3 significant figures.\MarkOne \AnswerBox
  \item A room has length \(5.2\) m and width \(3.8\) m, each correct to 1 decimal place. Find the upper bound for its perimeter.\MarkOne \AnswerBox
  \item A machine part is measured as \(12.0\) mm correct to the nearest \(0.1\) mm. A tolerance is \(\pm0.4\%\). Is this measurement accuracy sufficient?\MarkOne \AnswerBox
\end{enumerate}
