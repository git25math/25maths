\SectionHeader{Section A: Rounding Precision}{Decimal places, significant figures, and error intervals}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Round \(0.07846\) to 3 significant figures.\MarkOne \AnswerBox
  \item Round \(14872\) to 2 significant figures.\MarkOne \AnswerBox
  \item Round \(53.196\) to 1 decimal place.\MarkOne \AnswerBox
  \item Round \(7.84\) to 1 significant figure.\MarkOne \AnswerBox
  \item A length \(L\) is measured as \(6.3\) cm correct to 1 decimal place. Write the error interval for \(L\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Numerical Estimation}{Estimating calculations and percentages efficiently}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A value is \(920\) correct to 2 significant figures. Find the upper bound.\MarkOne \AnswerBox
  \item Estimate \(\frac{48.9\times0.198}{9.81}\) by rounding each number to 1 significant figure.\MarkOne \AnswerBox
  \item Estimate \(\sqrt{399}\).\MarkOne \AnswerBox
  \item Without a calculator, estimate \(\frac{197}{489}\) as a percentage.\MarkOne \AnswerBox
  \item A rectangle has length \(12.4\) cm and width \(8.7\) cm, each measured to the nearest \(0.1\) cm. Find the upper bound for its area.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Bounds Problems}{Upper/lower bounds in multi-step contexts}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Using Q10, find the lower bound for the area.\MarkOne \AnswerBox
  \item Mass \(A=3.46\) kg correct to 2 decimal places and mass \(B=1.9\) kg correct to 1 decimal place. Find the greatest possible value of \(A+B\).\MarkOne \AnswerBox
  \item Using Q12, find the smallest possible value of \(A-B\).\MarkOne \AnswerBox
  \item Estimate \(\frac{19.8^2-10.1^2}{5.02}\) by rounding each value to 1 significant figure.\MarkOne \AnswerBox
  \item Given \(x=7.2\) and \(y=3.4\), both correct to 1 decimal place, find the greatest possible value of \(\frac{x}{y}\), giving your answer to 2 decimal places.\MarkOne \AnswerBox
\end{enumerate}
