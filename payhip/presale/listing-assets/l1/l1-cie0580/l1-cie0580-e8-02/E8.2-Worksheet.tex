\SectionHeader{Section A: Relative Frequency Fluency}{Estimating probabilities from observed data}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item In 200 spins of a spinner, red appears 86 times. Find relative frequency of red.\MarkOne \AnswerBox
  \item In 500 coin tosses, heads appears 243 times. Find relative frequency of heads.\MarkOne \AnswerBox
  \item In 30 days, it rained on 18 days. Estimate probability of rain on a random day.\MarkOne \AnswerBox
  \item If estimated probability of success is 0.43, find expected successes in 200 trials.\MarkOne \AnswerBox
  \item If probability of defect is 0.04, find expected defective items in 750 items.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Expected Frequency Calculations}{Using E = np in practical contexts}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A relative frequency is 0.28 from 250 trials. Estimate number of successes.\MarkOne \AnswerBox
  \item In a simulation, 62 successes occur in 80 trials. Estimate expected successes in 500 trials.\MarkOne \AnswerBox
  \item If \(P(\text{win})=0.12\), find expected wins in 150 games.\MarkOne \AnswerBox
  \item If expected number of winners is 27 and \(P(\text{win})=0.18\), find number of games played.\MarkOne \AnswerBox
  \item In a sample of 400 households, 92 have cats. Estimate number with cats in a town of 12500 households.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Data-Based Predictions}{Scaling sample proportions to larger populations}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item In a quality sample, 36 of 450 items are defective. Estimate expected numbers of defective and non-defective items in a batch of 3200.\MarkOne \AnswerBox
  \item Over 5 days, a machine made \(120,150,130,140,160\) items with defects \(6,9,7,8,10\). Use pooled relative frequency to estimate defects in the next 900 items (nearest whole number).\MarkOne \AnswerBox
  \item A game has 47 wins in 125 plays. For 900 plays, estimate the expected number of wins and the probability of not winning.\MarkOne \AnswerBox
  \item Given expected successes are 84 and estimated probability of success is 0.35, determine the number of trials.\MarkOne \AnswerBox
  \item A travel survey finds 18\% use bus and 42\% use car; the rest use other transport. Estimate the number using other transport in a city of 15000 commuters.\MarkOne \AnswerBox
\end{enumerate}
