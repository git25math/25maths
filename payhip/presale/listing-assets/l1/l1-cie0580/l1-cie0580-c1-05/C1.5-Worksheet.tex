% ---- SECTION A
\SectionHeader{Section A: Integers, Decimals \& Negatives}{Ordering whole numbers and decimals, including negatives}{8--10 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Order from smallest to largest: \(-3,\ 1,\ -7,\ 0,\ 5\).\MarkOne \AnswerBox
  \item Order from greatest to smallest: \(4.2,\ 4.02,\ 4.12,\ 4.21\).\MarkOne \AnswerBox
  \item Which number is greater: \(-0.6\) or \(-0.06\)?\MarkOne \AnswerBox
  \item Order from smallest to largest: \(-2.5,\ -2.05,\ -2.15\).\MarkOne \AnswerBox
  \item Write \(<,\ >,\) or \(=\) to make this true: \(1.07\ \square\ 1.7\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Fractions, Decimals \& Percentages}{Comparing and ordering mixed number forms}{6--8 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Order from smallest to largest: \(\frac{1}{2},\ 0.45,\ 48\%,\ \frac{3}{5}\).\MarkOne \AnswerBox
  \item Which is greater: \(\frac{7}{8}\) or \(0.86\)?\MarkOne \AnswerBox
  \item Order from greatest to smallest: \(0.3,\ \frac{1}{4},\ 32\%,\ 0.305\).\MarkOne \AnswerBox
  \item Write \(<,\ >,\) or \(=\) to make this true: \(\frac{2}{5}\ \square\ 0.4\).\MarkOne \AnswerBox
  \item Order from smallest to largest: \(1.2,\ 118\%,\ \frac{5}{4},\ 1.19\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Number Line \& Real-life Ordering}{Comparing values in context and placing numbers accurately}{7--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item On a number line, place these values from left to right: \(-1.5,\ 2,\ -3.2\).\MarkOne \AnswerBox
  \item The temperature is \(-2^\circ\mathrm{C}\) at 6 am and \(3^\circ\mathrm{C}\) at noon. Which temperature is lower?\MarkOne \AnswerBox
  \item Three race times are \(12.8\) s, \(12.75\) s, and \(12.805\) s. Which is the fastest time?\MarkOne \AnswerBox
  \item Order from greatest to smallest: \(\$4.50,\ \$4.05,\ \$4.15\).\MarkOne \AnswerBox
  \item Order from smallest to largest: \(-0.8,\ -\frac{3}{4},\ -0.7\).\MarkOne \AnswerBox
\end{enumerate}
