\SectionHeader{Section A: Perpendicular Gradient Rule}{Negative reciprocal gradients}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Find gradient of a line perpendicular to \(y=3x-2\).\MarkOne \AnswerBox
  \item Find gradient of a line perpendicular to \(y=-\frac{1}{2}x+7\).\MarkOne \AnswerBox
  \item Find equation of line perpendicular to \(y=4x+1\) through \((0,5)\).\MarkOne \AnswerBox
  \item Find equation of line perpendicular to \(y=-3x+2\) through \((6,-1)\).\MarkOne \AnswerBox
  \item Are \(y=2x+1\) and \(y=-\frac{1}{2}x+4\) perpendicular?\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Perpendicular Equation Building}{Constructing lines with right-angle conditions}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Are lines \(3x+6y=12\) and \(2x-y=5\) perpendicular?\MarkOne \AnswerBox
  \item Find equation of line perpendicular to \(x=7\) through \((2,3)\).\MarkOne \AnswerBox
  \item Find equation of line perpendicular to \(y=-4\) through \((5,-1)\).\MarkOne \AnswerBox
  \item A line through \((1,2)\) and \((5,4)\) has gradient \(m\). Find gradient of a line perpendicular to it.\MarkOne \AnswerBox
  \item Find equation of line perpendicular to \(y=0.2x+9\) through \((-10,0)\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Right-Angle Applications}{Coordinate checks and geometric proofs by slope}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Find \(k\) if \(y=(k-1)x+6\) is perpendicular to \(y=\frac{1}{4}x-3\).\MarkOne \AnswerBox
  \item Find equation of line perpendicular to \(y=2x-5\) passing through the x-intercept of \(y=2x-5\).\MarkOne \AnswerBox
  \item For \(A(0,0)\), \(B(4,6)\), find equation of line through \(A\) perpendicular to \(AB\).\MarkOne \AnswerBox
  \item If slope \(m=\frac{5}{3}\), find slope of a perpendicular line.\MarkOne \AnswerBox
  \item Determine whether \(\angle BAC\) is a right angle for \(A(1,1),B(5,3),C(3,-3)\).\MarkOne \AnswerBox
\end{enumerate}
