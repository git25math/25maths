\SectionHeader{Section A: Independent Event Rules}{Multiplication and union formulas}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item If \(P(A)=0.4\), \(P(B)=0.5\), and \(A,B\) are independent, find \(P(A\cap B)\).\MarkOne \AnswerBox
  \item Using Q1 values, find \(P(A\cup B)\).\MarkOne \AnswerBox
  \item Two fair coins are tossed. Find probability of exactly one head.\MarkOne \AnswerBox
  \item Two fair dice are rolled. Find probability that the sum is at least 10.\MarkOne \AnswerBox
  \item A bag has 4 red and 6 blue counters. Two are drawn with replacement. Find \(P(\text{red then red})\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: With/Without Replacement}{Tree-diagram style reasoning for dependent events}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Using same bag, two are drawn without replacement. Find \(P(\text{red then red})\).\MarkOne \AnswerBox
  \item If \(P(A)=0.65\), \(P(B)=0.4\), and \(P(A'\cap B)=0.12\), find \(P(A\cap B)\).\MarkOne \AnswerBox
  \item Choose two numbers from 1 to 8 without replacement. Find probability both are even.\MarkOne \AnswerBox
  \item If \(P(A)=0.55\), \(P(B)=0.48\), \(P(A\cap B)=0.21\), find \(P(A\text{ only})\).\MarkOne \AnswerBox
  \item Using Q9 values, find \(P(\text{neither }A\text{ nor }B)\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Combined Event Problems}{Complements, Venn links, and mixed constraints}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A fair coin and a fair spinner numbered 1 to 5 are used. Find \(P(\text{head and even number})\).\MarkOne \AnswerBox
  \item In a class of 30, 18 study Maths, 12 study Science, 5 study both. Find probability a student studies at least one.\MarkOne \AnswerBox
  \item A batch has 3 defective and 17 good items. Two are chosen without replacement. Find \(P(\text{at least one defective})\).\MarkOne \AnswerBox
  \item If \(P(A)=0.7\) and \(P(B|A)=0.3\), find \(P(A\cap B)\).\MarkOne \AnswerBox
  \item If \(P(A)=p\), \(P(B)=0.6\), \(A,B\) independent, and \(P(A\cup B)=0.76\), find \(p\).\MarkOne \AnswerBox
\end{enumerate}
