\SectionHeader{Section A: Conversion Accuracy}{Converting between ordinary numbers and standard form}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Write \(6\,840\,000\) in standard form.\MarkOne \AnswerBox
  \item Write \(0.0000542\) in standard form.\MarkOne \AnswerBox
  \item Write \(9.37\times10^4\) as an ordinary number.\MarkOne \AnswerBox
  \item Write \(4.08\times10^{-3}\) as an ordinary number.\MarkOne \AnswerBox
  \item Calculate \((6\times10^5)(3\times10^2)\) in standard form.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Standard Form Calculations}{Multiply, divide, add, and subtract in standard form}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Calculate \(\frac{8.4\times10^7}{2.1\times10^3}\) in standard form.\MarkOne \AnswerBox
  \item Calculate \((7.2\times10^{-4})+(5.6\times10^{-4})\) in standard form.\MarkOne \AnswerBox
  \item Calculate \((9.1\times10^6)-(2.7\times10^6)\) in standard form.\MarkOne \AnswerBox
  \item Calculate \((4\times10^8)(5\times10^{-6})\) in standard form.\MarkOne \AnswerBox
  \item Calculate \(\frac{6.3\times10^{-4}}{9\times10^2}\) in standard form.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Applications}{Real-world interpretation and algebra with powers of ten}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A bacterium has mass \(3.2\times10^{-6}\) g. There are \(7.5\times10^8\) bacteria. Find the total mass in grams in standard form.\MarkOne \AnswerBox
  \item Light travels at \(3.0\times10^8\) m/s for \(2.5\times10^{-3}\) s. Find the distance in metres in standard form.\MarkOne \AnswerBox
  \item A rectangle has length \(4.2\times10^3\) m and width \(6.0\times10^{-2}\) m. Find its area in standard form.\MarkOne \AnswerBox
  \item Express \(9.6\times10^{12}:3.0\times10^9\) as a simplest ratio of integers.\MarkOne \AnswerBox
  \item Given \(4.8\times10^p=0.0048\), find \(p\).\MarkOne \AnswerBox
\end{enumerate}
