% ---- SECTION A
\SectionHeader{Section A: Data-Type Basics}{Identify standard statistical categories}{8--10 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Is "eye color" qualitative or quantitative data?\MarkOne \AnswerBox
  \item Is "number of siblings" discrete or continuous data?\MarkOne \AnswerBox
  \item Is "height of students" discrete or continuous data?\MarkOne \AnswerBox
  \item Is "favorite fruit" qualitative or quantitative data?\MarkOne \AnswerBox
  \item Which chart type is usually used for categorical (qualitative) data: bar chart or histogram?\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Classifying Real Datasets}{Use precise labels for common variables}{6--8 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Race times recorded as 12.4 s, 11.9 s, 12.1 s are discrete or continuous?\MarkOne \AnswerBox
  \item Letter grades \(A,B,C,\ldots\) are qualitative or quantitative?\MarkOne \AnswerBox
  \item Number of cars passing a point each minute is discrete or continuous?\MarkOne \AnswerBox
  \item Weights of apples are discrete or continuous?\MarkOne \AnswerBox
  \item Data organized into class intervals is often shown with which chart type?\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Representation Reasoning}{Choose displays and justify data type logic}{7--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Which chart best shows parts of a whole for categories?\MarkOne \AnswerBox
  \item Which graph type is best for tracking change over time?\MarkOne \AnswerBox
  \item True or False: Histograms are used for grouped continuous data.\MarkOne \AnswerBox
  \item Give one example of a discrete quantitative variable.\MarkOne \AnswerBox
  \item If a variable can take any value in an interval, what type of data is it?\MarkOne \AnswerBox
\end{enumerate}
