% ---- SECTION A
\SectionHeader{Section A: Percentage Basics}{Percentage of amount, decimal/percent conversion, discount}{8--10 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Find \(15\%\) of \(240\).\MarkOne \AnswerBox
  \item Find \(8\%\) of \(350\).\MarkOne \AnswerBox
  \item Write \(0.65\) as a percentage.\MarkOne \AnswerBox
  \item Write \(42\%\) as a decimal.\MarkOne \AnswerBox
  \item A jacket costs \$80. It is discounted by \(25\%\). Find the sale price.\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Percentage Change}{Increase, decrease, percent change in context}{6--8 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A phone costs \$120 and its price increases by \(10\%\). Find the new price.\MarkOne \AnswerBox
  \item The price of a bike falls from \$250 to \$225. Find the percentage decrease.\MarkOne \AnswerBox
  \item A town's population rises from 18\,000 to 19\,260. Find the percentage increase.\MarkOne \AnswerBox
  \item A shirt price rises from \$35 to \$42. Find the percentage increase.\MarkOne \AnswerBox
  \item A bill is reduced from \$48.00 to \$40.80. Find the percentage decrease.\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Reverse Percentage \& Context}{Original value, profit/loss, simple interest style}{7--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item After a \(20\%\) discount, a bag costs \$64. Find the original price.\MarkOne \AnswerBox
  \item After a \(12\%\) increase, monthly rent becomes \$224. Find the original rent.\MarkOne \AnswerBox
  \item A shop buys an item for \$45 and sells it for \$54. Find the percentage profit.\MarkOne \AnswerBox
  \item A loan of \$500 is charged simple interest at \(4\%\) per year for 3 years. Find the total interest.\MarkOne \AnswerBox
  \item The price including \(15\%\) VAT is \$92. Find the price before VAT.\MarkOne \AnswerBox
\end{enumerate}
