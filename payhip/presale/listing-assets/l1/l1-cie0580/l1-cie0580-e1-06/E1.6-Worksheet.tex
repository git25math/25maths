\SectionHeader{Section A: Numerical Fluency with Operations}{Accurate arithmetic with integers, decimals, fractions}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Calculate \(-14+27-9\).\MarkOne \AnswerBox
  \item Evaluate \(3.6\times0.45\).\MarkOne \AnswerBox
  \item Work out \(\frac{7}{8}-\frac{5}{12}\).\MarkOne \AnswerBox
  \item Calculate \(5.04\div0.12\).\MarkOne \AnswerBox
  \item Evaluate \((-3)^2-4(-2)\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Multi-step Operation Chains}{BIDMAS, signed numbers, and mixed forms}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Evaluate \(18-3[2+(5-9)]\).\MarkOne \AnswerBox
  \item Calculate \(\left(\frac{3}{4}\div\frac{5}{6}\right)+\frac{1}{10}\).\MarkOne \AnswerBox
  \item Evaluate \(2.5^2-3.4\div0.2\).\MarkOne \AnswerBox
  \item Find the value of \(\frac{48-3\times7}{3^2}\).\MarkOne \AnswerBox
  \item A number is multiplied by 1.5 then 8 is subtracted to give 22. Find the number.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Applied Arithmetic}{Context modelling using four operations}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A factory produces 2400 bolts. \(12.5\%\) fail quality check. The remaining bolts are packed equally into boxes of 36. Determine the number of full boxes and bolts left over.\MarkOne \AnswerBox
  \item Given \(A=(3.6-1.25)\times(4.8\div0.6)\) and \(B=2\frac{3}{4}+\frac{5}{8}\), determine \(A-B\).\MarkOne \AnswerBox
  \item A recipe for 8 servings uses \(1\frac{1}{4}\) kg flour and 0.6 kg sugar. Given 3.5 kg flour and 2.1 kg sugar, determine the maximum number of complete servings.\MarkOne \AnswerBox
  \item A taxi fare is \pounds4.80 plus \pounds1.35 per km. Then \(8\%\) tax is added to the fare. Determine the total cost for 14 km.\MarkOne \AnswerBox
  \item A tank is \(\frac{3}{5}\) full of 900 L. Then 120 L is used, and \(18\%\) of the remaining amount is refilled. Determine the final volume of water.\MarkOne \AnswerBox
\end{enumerate}
