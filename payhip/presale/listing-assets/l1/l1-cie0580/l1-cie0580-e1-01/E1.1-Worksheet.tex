\SectionHeader{Section A: Number Classification Precision}{Natural, integer, rational, irrational, and real sets}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item For the number \(-15\), state the smallest set it belongs to: \(\mathbb{N},\mathbb{Z},\mathbb{Q},\mathbb{R}\).\MarkOne \AnswerBox
  \item State the smallest set containing \(\sqrt{49}\).\MarkOne \AnswerBox
  \item Determine whether \(\pi-3\) is rational or irrational.\MarkOne \AnswerBox
  \item Write \(0.\overline{27}\) as a fraction in simplest form.\MarkOne \AnswerBox
  \item Determine whether \(\sqrt{50}-5\) is rational or irrational.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Converting and Comparing Forms}{Recurring decimals, exact forms, and order reasoning}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Write \(0.12\overline{3}\) as an exact fraction.\MarkOne \AnswerBox
  \item Express \(1.2\times10^{-3}\) as \(\frac{a}{b}\) in simplest form.\MarkOne \AnswerBox
  \item Order these numbers from smallest to largest: \(-\sqrt{2},\,-1.41,\,-\frac{7}{5}\).\MarkOne \AnswerBox
  \item Evaluate \((2+\sqrt{3})(2-\sqrt{3})\) and state its number type.\MarkOne \AnswerBox
  \item Find an integer \(n\) such that \(n<\sqrt{50}<n+1\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Number Reasoning}{Justification, bounds, and contextual interpretation}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A square has side length \(\sqrt{18}\) m. Write its perimeter in simplified surd form.\MarkOne \AnswerBox
  \item A value starts at \(12\) and decreases by \(3.5\) each hour for 6 hours. Find the final value.\MarkOne \AnswerBox
  \item Give one irrational number strictly between 2 and 3.\MarkOne \AnswerBox
  \item The calculator shows \(0.3333\ldots\). State its exact value.\MarkOne \AnswerBox
  \item Is \(\frac{\sqrt{8}}{\sqrt{2}}\) rational or irrational?\MarkOne \AnswerBox
\end{enumerate}
