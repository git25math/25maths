\SectionHeader{Section A: 3D Pythagoras Foundations}{Space diagonals and planar projections}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A cuboid has dimensions 3 cm, 4 cm, 12 cm. Find its space diagonal.\MarkOne \AnswerBox
  \item A cuboid has dimensions 6 cm, 8 cm, 24 cm. Find its space diagonal.\MarkOne \AnswerBox
  \item A right square prism has base side 10 cm and height 24 cm. Find distance from top centre of one face to opposite base corner.\MarkOne \AnswerBox
  \item A cuboid has dimensions 5 cm, 12 cm, 13 cm. Find angle between its space diagonal and the base plane.\MarkOne \AnswerBox
  \item For the cuboid in Q4, find angle between space diagonal and the vertical edge.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: 3D Trigonometric Angles}{Angles with planes, lines, and verticals}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A cuboid has base \(9\text{ cm}\times12\text{ cm}\) and space diagonal 20 cm. Find height.\MarkOne \AnswerBox
  \item A room is \(8\text{ m}\times6\text{ m}\times3\text{ m}\). Find corner-to-corner distance.\MarkOne \AnswerBox
  \item A guy wire runs from top of 15 m pole to a ground point 9 m east and 12 m north of base. Find wire length.\MarkOne \AnswerBox
  \item A cube has edge length 7 cm. Find body diagonal in exact form.\MarkOne \AnswerBox
  \item A cuboid has dimensions 4 cm, 4 cm, 10 cm. Find angle between space diagonal and vertical edge.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended 3D Problem Solving}{Multi-step distance-angle modelling in solids}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A ramp rises 5 m over horizontal plan distance 12 m. Find ramp length.\MarkOne \AnswerBox
  \item For Q11 ramp, find angle to horizontal.\MarkOne \AnswerBox
  \item A square-based pyramid has base side 6 cm and vertical height 8 cm. Find slant edge from apex to base corner.\MarkOne \AnswerBox
  \item For Q13 pyramid, find angle between this slant edge and base plane.\MarkOne \AnswerBox
  \item Find distance between \(A(0,0,0)\) and \(B(4,7,1)\) in 3D.\MarkOne \AnswerBox
\end{enumerate}
