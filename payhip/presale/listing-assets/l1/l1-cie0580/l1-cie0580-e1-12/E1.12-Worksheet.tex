\SectionHeader{Section A: Core Rate Calculations}{Speed, density, and unit cost fundamentals}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A car travels 180 km in 2.5 h. Find its average speed.\MarkOne \AnswerBox
  \item A tank contains 96 L of water and empties at 12 L/min. How long does it take to empty?\MarkOne \AnswerBox
  \item 4 kg of apples cost \pounds10.60. Find the unit cost per kg.\MarkOne \AnswerBox
  \item A sample has mass 540 g and volume \(200\text{ cm}^3\). Find its density.\MarkOne \AnswerBox
  \item A cyclist travels at \(7.5\text{ m/s}\). Convert this speed to km/h.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Multi-Step Rate Problems}{Converting units and chaining rate relationships}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A train travels at 90 km/h for 35 minutes. Find the distance travelled.\MarkOne \AnswerBox
  \item A parcel has density \(0.8\text{ g/cm}^3\) and volume \(2500\text{ cm}^3\). Find its mass.\MarkOne \AnswerBox
  \item A tap fills 18 L in 45 s. Find the rate in L/min.\MarkOne \AnswerBox
  \item 12 notebooks cost \pounds15.60. Find the cost of 25 notebooks at the same rate.\MarkOne \AnswerBox
  \item A liquid has mass 1.44 kg and density \(1.2\text{ g/cm}^3\). Find its volume in \(\text{cm}^3\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Variation and Rates}{Direct/inverse variation in realistic scenarios}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item For a fixed distance, time \(t\) is inversely proportional to speed \(s\). If \(t=3\) h when \(s=50\) km/h, find \(t\) when \(s=75\) km/h.\MarkOne \AnswerBox
  \item Cost \(C\) is directly proportional to mass \(m\). If 3.5 kg costs \pounds8.40, find the cost of 5.2 kg.\MarkOne \AnswerBox
  \item A printer produces 540 pages in 12 minutes at a constant rate. How long for 900 pages?\MarkOne \AnswerBox
  \item A journey of 120 km has two stages: 80 km at 64 km/h and 40 km at 40 km/h. Find the average speed for the whole journey.\MarkOne \AnswerBox
  \item A liquid has density \(1.25\text{ g/cm}^3\). Three containers each hold 0.4 L and are filled. Find the total mass in kg.\MarkOne \AnswerBox
\end{enumerate}
