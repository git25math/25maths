% ---- SECTION A
\SectionHeader{Section A: Algebra Basics}{Variables, terms, and expression meaning}{8--10 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item In \(5x + 3\), what is the coefficient of \(x\)?\MarkOne \AnswerBox
  \item Write an expression for "a number \(n\) decreased by 7".\MarkOne \AnswerBox
  \item If \(x=4\), evaluate \(3x+2\).\MarkOne \AnswerBox
  \item If \(a=2\) and \(b=5\), evaluate \(a+b\).\MarkOne \AnswerBox
  \item Simplify \(4x+3x\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Simplifying \& Substituting}{Collect like terms and evaluate expressions}{6--8 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Simplify \(5y-2y+6\).\MarkOne \AnswerBox
  \item Simplify \(3(a+2)+a\).\MarkOne \AnswerBox
  \item If \(m=7\), evaluate \(2m-5\).\MarkOne \AnswerBox
  \item If \(p=6\), evaluate \(p^2\).\MarkOne \AnswerBox
  \item Simplify \(2x+5-3x+1\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Simple Equations}{Solve one-step and two-step equations}{7--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Solve \(x+6=14\).\MarkOne \AnswerBox
  \item Solve \(3x=21\).\MarkOne \AnswerBox
  \item Solve \(\frac{x}{4}=5\).\MarkOne \AnswerBox
  \item Solve \(2x+3=11\).\MarkOne \AnswerBox
  \item The sum of a number \(n\) and 8 is 17. Find \(n\).\MarkOne \AnswerBox
\end{enumerate}
