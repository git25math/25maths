\SectionHeader{Section A: Core Average Calculations}{Mean, median, mode, and range from raw data}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Find the mean of \(4,7,9,10,15\).\MarkOne \AnswerBox
  \item Find the median of \(3,5,6,8,12,14\).\MarkOne \AnswerBox
  \item Find the mode of \(2,4,4,5,7,7,7,9\).\MarkOne \AnswerBox
  \item Find the range of \(18,11,24,9,20\).\MarkOne \AnswerBox
  \item Find weighted mean of scores 60, 80, 90 with weights 2, 3, 1.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Frequency and Weighted Means}{Using frequencies and weights correctly}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item For values \(1,2,3,4\) with frequencies \(2,5,4,1\), find mean.\MarkOne \AnswerBox
  \item Grouped data \(0\!-\!10:3,\ 10\!-\!20:5,\ 20\!-\!30:2\). Estimate mean using class midpoints.\MarkOne \AnswerBox
  \item The mean of 8 numbers is 12. One number is removed and it is 20. Find new mean.\MarkOne \AnswerBox
  \item If range is 32 and minimum is 9, find maximum.\MarkOne \AnswerBox
  \item The mean of 5 numbers is 18. Four are 16, 20, 19, 17. Find the fifth.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Summary Reasoning}{Missing-value and combined-mean problems}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Group A: 20 students mean 68. Group B: 30 students mean 74. Find combined mean.\MarkOne \AnswerBox
  \item If numbers are \(a,\ a+2,\ a+4\) and their mean is 18, find \(a\).\MarkOne \AnswerBox
  \item Find median of \(2,3,3,5,8,8,8,9,10\).\MarkOne \AnswerBox
  \item For ordered data \(4,6,7,9,11,13,16,18\), find IQR.\MarkOne \AnswerBox
  \item Grouped data \(0\!-\!5:4,\ 5\!-\!10:6,\ 10\!-\!15:10\). Estimate mean using midpoints.\MarkOne \AnswerBox
\end{enumerate}
