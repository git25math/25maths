\SectionHeader{Section A: Reading Tables and Summaries}{Frequency, cumulative counts, and class interpretation}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Data values \(1,2,3,4\) have frequencies \(4,7,5,4\). Find total frequency.\MarkOne \AnswerBox
  \item For the data in Q1, find the median value.\MarkOne \AnswerBox
  \item Grouped data: \(0\!-\!10:3,\ 10\!-\!20:9,\ 20\!-\!30:6,\ 30\!-\!50:8\). State the modal class.\MarkOne \AnswerBox
  \item Using Q3, find which class has highest frequency density.\MarkOne \AnswerBox
  \item A box plot has minimum 12, \(Q_1=18\), median 24, \(Q_3=31\), maximum 40. Find IQR.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Interpreting Box/Histogram Features}{Range, IQR, density, and spread conclusions}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Using Q5 values, find range.\MarkOne \AnswerBox
  \item For 80 ordered values, where is the median position?\MarkOne \AnswerBox
  \item For 80 ordered values, at what cumulative frequency is the upper quartile read?\MarkOne \AnswerBox
  \item If a frequency table has total 50 and class \(0\!-\!20\) has frequency 12, estimate probability a value is in \(0\!-\!20\).\MarkOne \AnswerBox
  \item If \(Q_1=35\) and \(Q_3=55\), find the upper outlier fence \(Q_3+1.5\text{IQR}\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Interpretation Judgements}{Distribution shape and inference limits}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Cumulative frequencies at boundaries 10,20,30,40 are 4,11,19,30. Find frequency in class \(20\!-\!30\).\MarkOne \AnswerBox
  \item A histogram bar has class width 5 and frequency density 3.2. Find its frequency.\MarkOne \AnswerBox
  \item A histogram bar has class width 12 and frequency density 1.5. Find its frequency.\MarkOne \AnswerBox
  \item If mean is noticeably greater than median, is the distribution likely positively or negatively skewed?\MarkOne \AnswerBox
  \item For right-skewed data, order mean, median, mode from largest to smallest.\MarkOne \AnswerBox
\end{enumerate}
