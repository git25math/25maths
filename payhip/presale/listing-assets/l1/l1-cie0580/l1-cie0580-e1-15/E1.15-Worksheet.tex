\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Convert \(2\) h \(45\) min to minutes.\MarkOne \AnswerBox
  \item Convert \(3.6\) hours to hours and minutes.\MarkOne \AnswerBox
  \item Find the duration from 07:38 to 11:05.\MarkOne \AnswerBox
  \item A film starts at 18:25 and ends at 20:47. Find its duration.\MarkOne \AnswerBox
  \item A bus leaves at 23:40 and arrives at 01:15 the next day. Find the journey time.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Time in Multi-Step Contexts}{Schedules, timetables, and journey calculations}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A flight leaves London at 21:50 and lasts 8 h 25 min. Dubai is 4 hours ahead of London. Find the local arrival time in Dubai.\MarkOne \AnswerBox
  \item A runner's pace is 4 min 48 s per km. Find the time for 12 km.\MarkOne \AnswerBox
  \item A task starts at 08:40. It lasts 2 h 45 min, then there is a 20 min break, then 1 h 30 min more work. Find finishing time.\MarkOne \AnswerBox
  \item A car travels at 72 km/h for 1 h 45 min. Find the distance travelled.\MarkOne \AnswerBox
  \item A journey of 210 km takes 3 h 30 min. Find the average speed.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Time Applications}{Time zones, rate-time links, and combined intervals}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A clock gains 5 min every 12 h. It is correct at 08:00 Monday. What time will it show when true time is 08:00 Wednesday?\MarkOne \AnswerBox
  \item An event starts at 13:35. Session 1 lasts 2 h 18 min, then a 35 min break, then Session 2 lasts 1 h 42 min. Find finishing time.\MarkOne \AnswerBox
  \item Time in UTC+2 is 18:30 on Friday. What is the time in UTC-5?\MarkOne \AnswerBox
  \item A car travels 80 km at 80 km/h, then 40 km at 40 km/h. Find the average speed for the whole journey.\MarkOne \AnswerBox
  \item At constant speed, 240 km takes 3 h 12 min. How long will 375 km take?\MarkOne \AnswerBox
\end{enumerate}
