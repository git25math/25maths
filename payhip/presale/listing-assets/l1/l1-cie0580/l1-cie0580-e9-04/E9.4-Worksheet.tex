\SectionHeader{Section A: Chart Construction Numbers}{Pie angles, class midpoints, and scale reading}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item In a pie chart of total 200 items, category A has frequency 50. Find its sector angle.\MarkOne \AnswerBox
  \item In a pie chart of total 360 items, category B has frequency 72. Find its sector angle.\MarkOne \AnswerBox
  \item In a pie chart of total 240 items, a sector angle is \(54^\circ\). Find the category frequency.\MarkOne \AnswerBox
  \item A sector angle is \(120^\circ\). What fraction of the full pie does it represent?\MarkOne \AnswerBox
  \item Find midpoint of class interval \(20\!-\!30\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Diagram Interpretation}{Frequency polygons, box plots, and stems}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A histogram class has width 10 and density 2.8. Find its frequency.\MarkOne \AnswerBox
  \item A histogram class has frequency 45 and width 15. Find its density.\MarkOne \AnswerBox
  \item For cumulative frequency plotting, which class boundary is used for class \(30\!-\!40\)?\MarkOne \AnswerBox
  \item In a box plot with \(Q_1=28\), median 36, \(Q_3=44\), find box length.\MarkOne \AnswerBox
  \item In the same box plot with maximum 58, find upper whisker length.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Chart Selection and Analysis}{Choosing best representation and extracting values}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A pie chart sector angle is \(48^\circ\) out of total frequency 150. Find sector frequency.\MarkOne \AnswerBox
  \item Best chart for composition percentages by category?\MarkOne \AnswerBox
  \item Best chart for grouped continuous data with unequal class widths?\MarkOne \AnswerBox
  \item In a frequency polygon, what x-coordinate is used for class \(12\!-\!18\)?\MarkOne \AnswerBox
  \item If stem-and-leaf key is \(4|7=47\), what value is \(6|3\)?\MarkOne \AnswerBox
\end{enumerate}
