\SectionHeader{Section A: Direct and Inverse Proportion}{Forming proportional models and solving quickly}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item \(y\) is directly proportional to \(x\). If \(y=18\) when \(x=6\), find \(y\) when \(x=14\).\MarkOne \AnswerBox
  \item \(p\) is inversely proportional to \(q\). If \(p=12\) when \(q=5\), find \(p\) when \(q=8\).\MarkOne \AnswerBox
  \item Cost \(C\) is directly proportional to mass \(m\). If 4 kg costs \pounds9.60, find the cost of 7.5 kg.\MarkOne \AnswerBox
  \item Time \(t\) is inversely proportional to number of workers \(n\). If 6 workers take 15 days, how many days for 10 workers?\MarkOne \AnswerBox
  \item On a map of scale \(1:50000\), a road is 8.4 cm long. Find the real length in km.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Proportion in Context}{Scale, rates, recipes, and exchange setups}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item In a recipe, flour:sugar \(=5:2\). If flour is 350 g, find sugar mass.\MarkOne \AnswerBox
  \item \(y\propto x^2\). If \(y=45\) when \(x=3\), find \(y\) when \(x=7\).\MarkOne \AnswerBox
  \item \(y\propto\frac{1}{\sqrt{x}}\). If \(y=10\) when \(x=25\), find \(y\) when \(x=4\).\MarkOne \AnswerBox
  \item Two similar figures have scale factor 3 (small to large). If small area is \(12\text{ cm}^2\), find large area.\MarkOne \AnswerBox
  \item Two similar solids have linear scale factor 2.5 (small to large). If small volume is \(64\text{ cm}^3\), find large volume.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Proportion Reasoning}{Compound proportion and similar-shape scaling}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item If \(1\text{ USD}=0.78\text{ GBP}\), convert 350 USD to GBP.\MarkOne \AnswerBox
  \item For a fixed distance, time is inversely proportional to speed. If a journey takes 2.5 h at 72 km/h, how long at 90 km/h?\MarkOne \AnswerBox
  \item Pressure \(p\) is inversely proportional to volume \(v\). If \(p=240\) when \(v=15\), find \(v\) when \(p=180\).\MarkOne \AnswerBox
  \item \(y=\frac{k}{x}\). If \(y=7\) when \(x=4\), find \(y\) when \(x=20\).\MarkOne \AnswerBox
  \item A paint mix has ratio red:blue:white \(=3:5:2\). If total paint is 4 L, find amount of blue paint.\MarkOne \AnswerBox
\end{enumerate}
