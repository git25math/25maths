\SectionHeader{Section A: Core Pythagoras Fluency}{Missing side calculations in right triangles}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A right triangle has legs 6 cm and 8 cm. Find the hypotenuse.\MarkOne \AnswerBox
  \item A right triangle has hypotenuse 13 cm and one leg 5 cm. Find the other leg.\MarkOne \AnswerBox
  \item A right triangle has legs 9 cm and 12 cm. Find the hypotenuse.\MarkOne \AnswerBox
  \item A right triangle has hypotenuse 17 cm and one leg 8 cm. Find the other leg.\MarkOne \AnswerBox
  \item A rectangle is \(7\text{ cm}\times24\text{ cm}\). Find its diagonal.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Coordinate and Context Applications}{Distance, diagonals, and right-angle verification}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Find distance between \((1,2)\) and \((7,10)\).\MarkOne \AnswerBox
  \item A square has side 5 cm. Find its diagonal in exact form.\MarkOne \AnswerBox
  \item A 13 m ladder stands 5 m from a wall. How high does it reach?\MarkOne \AnswerBox
  \item Check whether lengths 7, 24, 25 form a right triangle.\MarkOne \AnswerBox
  \item A field is \(30\text{ m}\times40\text{ m}\). Find shortest corner-to-corner distance.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Pythagoras Modelling}{Scaled forms and 3D extensions}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item In a right triangle, legs are \(3x\) and \(4x\), hypotenuse is 35. Find \(x\).\MarkOne \AnswerBox
  \item A cuboid has side lengths 3 cm, 4 cm, and 12 cm. Find space diagonal.\MarkOne \AnswerBox
  \item A right triangle has hypotenuse 20 cm and one leg 12 cm. Find other leg.\MarkOne \AnswerBox
  \item Two perpendicular vectors have magnitudes 9 and 40. Find magnitude of resultant.\MarkOne \AnswerBox
  \item A right triangle has legs \(a\) and \(b\). Write formula for hypotenuse \(c\).\MarkOne \AnswerBox
\end{enumerate}
