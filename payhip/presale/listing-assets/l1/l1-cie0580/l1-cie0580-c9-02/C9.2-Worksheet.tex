% ---- SECTION A
\SectionHeader{Section A: Interpreting Common Charts}{Read direct values and proportions}{8--10 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A frequency table has frequencies 2, 5, 7, and 6. Find the total frequency.\MarkOne \AnswerBox
  \item In a bar chart, which category has the highest bar?\MarkOne \AnswerBox
  \item A pie chart sector is \(90^\circ\). What fraction of the whole does it represent?\MarkOne \AnswerBox
  \item On a line graph, values rise from left to right. What does this indicate?\MarkOne \AnswerBox
  \item True or False: Histogram bars usually touch.\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Comparing and Summarizing Data}{Infer modal classes and differences}{6--8 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Class intervals \(0\text{--}10, 10\text{--}20, 20\text{--}30\) have frequencies 4, 9, 7. Find the modal class.\MarkOne \AnswerBox
  \item Cumulative frequencies are 3, 8, 15, 20. How many values are at or below the third class boundary?\MarkOne \AnswerBox
  \item A bar chart shows category \(A=12\) and \(B=18\). Find the difference.\MarkOne \AnswerBox
  \item In a pie chart, a sector of \(36^\circ\) represents what percentage?\MarkOne \AnswerBox
  \item A line graph value at Day 4 is 28 units. What is the recorded value on Day 4?\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Interpretation in Context}{Turn chart data into decisions}{7--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item In a travel survey of 50 students, 20 use bus. Find the proportion using bus.\MarkOne \AnswerBox
  \item When histogram class widths are unequal, what should be compared: frequency or frequency density?\MarkOne \AnswerBox
  \item True or False: A graph always shows the mean directly.\MarkOne \AnswerBox
  \item For 21 ordered data values, what position is the median?\MarkOne \AnswerBox
  \item Data values are 5, 7, 7, 9, 10, 12, 12, 14. Find the range.\MarkOne \AnswerBox
\end{enumerate}
