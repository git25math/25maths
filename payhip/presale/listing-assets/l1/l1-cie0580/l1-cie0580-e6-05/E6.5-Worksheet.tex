\SectionHeader{Section A: Sine/Cosine Rule Setup}{Choosing rule based on given data}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Find area of a triangle with sides 7 cm and 9 cm, included angle \(60^\circ\).\MarkOne \AnswerBox
  \item Find area of a triangle with sides 8 cm and 11 cm, included angle \(45^\circ\).\MarkOne \AnswerBox
  \item Sides 5 cm and 7 cm include angle \(120^\circ\). Find third side.\MarkOne \AnswerBox
  \item Triangle sides are 10 cm, 12 cm, 14 cm. Find angle opposite side 14 cm.\MarkOne \AnswerBox
  \item Given \(a=8,\ A=40^\circ,\ B=70^\circ\), find side \(b\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Area and Unknown Solving}{Using 1/2 ab sin C and rule combinations}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Given \(a=9,\ A=35^\circ,\ b=12\), find acute angle \(B\).\MarkOne \AnswerBox
  \item Sides 6 cm and 9 cm include angle \(30^\circ\). Find third side.\MarkOne \AnswerBox
  \item Find area with sides 12 cm and 15 cm, included angle \(110^\circ\).\MarkOne \AnswerBox
  \item If side opposite \(50^\circ\) is 10 cm, find side opposite \(65^\circ\).\MarkOne \AnswerBox
  \item Triangle sides are 7 cm, 10 cm, 12 cm. Find largest angle.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Triangle Modelling}{Multi-step side/angle/perimeter reasoning}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Sides 9 cm and 13 cm include angle \(75^\circ\). Find third side.\MarkOne \AnswerBox
  \item Triangle area is \(48\text{ cm}^2\) with sides 10 cm and 12 cm enclosing angle \(C\). Find acute \(C\).\MarkOne \AnswerBox
  \item Given \(a=15,\ A=48^\circ,\ b=19\), find acute angle \(B\).\MarkOne \AnswerBox
  \item Given sides 14 cm and 18 cm with opposite angle to side 14 cm equal to \(40^\circ\), find acute angle opposite 18 cm.\MarkOne \AnswerBox
  \item Sides 9 cm and 11 cm include angle \(70^\circ\). Find triangle perimeter.\MarkOne \AnswerBox
\end{enumerate}
