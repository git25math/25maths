\SectionHeader{Section A: Ordering Mixed Forms}{Fractions, decimals, percentages, negatives}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Order from smallest to largest: \(-1.7,\,-\frac{5}{3},\,-1.62\).\MarkOne \AnswerBox
  \item Order ascending: \(35\%,\,0.29,\,\frac{3}{10}\).\MarkOne \AnswerBox
  \item Insert \(<\) or \(>\): \(-\sqrt{5}\ \square\ -2.2\).\MarkOne \AnswerBox
  \item Order ascending: \(\frac{7}{12},\,0.57,\,58\%\).\MarkOne \AnswerBox
  \item Find the median of: \(2.4,\ 2.35,\ 2.5,\ 2.41,\ 2.39\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Bounds and Intervals}{Upper/lower bounds and ordering within ranges}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Write in ascending order: \(4.1\times10^{-3},\ 3.9\times10^{-3},\ 0.00405\).\MarkOne \AnswerBox
  \item Given \(a=1.73,\ b=\sqrt{3},\ c=1.731\), order \(a,b,c\) descending.\MarkOne \AnswerBox
  \item A length is measured as \(7.2\) cm to 1 d.p. Write the lower and upper bounds.\MarkOne \AnswerBox
  \item Number \(x\) is rounded to nearest integer as 12. Number \(y\) is rounded to nearest integer as 13. Write the greatest possible value of \(x-y\).\MarkOne \AnswerBox
  \item Order ascending: \(-0.45,\ -45\%,\ -\frac{9}{20}\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Number Comparison}{Ordering surds, standard form, and contextual values}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Three runners take \(12.48\) s, \(12.5\) s (to 1 d.p.), and \(12.47\) s. Which runner is definitely fastest?\MarkOne \AnswerBox
  \item Arrange in ascending order: \(\sqrt{7},\ 2.64,\ \frac{8}{3}\).\MarkOne \AnswerBox
  \item If \(p=0.4\overline{7}\), \(q=\frac{7}{15}\), \(r=46.5\%\), order \(p,q,r\) descending.\MarkOne \AnswerBox
  \item The masses are \(2.0\) kg (to 1 d.p.), \(2.06\) kg, and \(2.12\) kg. Which is certainly the largest?\MarkOne \AnswerBox
  \item Find a number between \(0.03\) and \(0.031\) that is greater than \(0.0305\).\MarkOne \AnswerBox
\end{enumerate}
