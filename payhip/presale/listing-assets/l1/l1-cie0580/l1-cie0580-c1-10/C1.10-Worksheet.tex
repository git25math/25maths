% ---- SECTION A
\SectionHeader{Section A: Rounding Bounds}{Find bounds from rounded values}{8--10 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A length is \(12\) cm correct to the nearest cm. Write the error interval for \(L\).\MarkOne \AnswerBox
  \item A mass is \(3.7\) kg correct to 1 decimal place. Write the error interval for \(m\).\MarkOne \AnswerBox
  \item A time is \(45\) minutes correct to the nearest 5 minutes. Write the error interval for \(t\).\MarkOne \AnswerBox
  \item A value is \(0.84\) correct to 2 decimal places. Write the error interval for \(x\).\MarkOne \AnswerBox
  \item A number is \(6200\) correct to 2 significant figures. Write the error interval for \(n\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Degree of Accuracy}{Use nearest unit, d.p., and s.f. correctly}{6--8 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A width is \(8.0\) cm correct to the nearest \(0.5\) cm. Write the error interval for \(w\).\MarkOne \AnswerBox
  \item \(y=2.46\) correct to the nearest \(0.01\). Write the upper bound of \(y\).\MarkOne \AnswerBox
  \item \(z=130\) correct to the nearest 10. Write the error interval for \(z\).\MarkOne \AnswerBox
  \item \(p=0.070\) correct to 3 decimal places. Write the lower bound of \(p\).\MarkOne \AnswerBox
  \item \(q=4.2\) correct to 2 significant figures. Write the upper bound of \(q\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Bounds in Context}{Apply bounds to length, area, speed, and cost}{7--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A rectangle has length \(8\) cm and width \(5\) cm, each correct to nearest cm. Find the lower bound of perimeter \(P\).\MarkOne \AnswerBox
  \item A rectangle has length \(12\) cm and width \(6\) cm, each correct to nearest cm. Find the upper bound of area \(A\).\MarkOne \AnswerBox
  \item Distance \(=240\) km (nearest 10 km), time \(=4\) h (nearest hour). Find the upper bound of speed.\MarkOne \AnswerBox
  \item Mass \(=2.5\) kg correct to 1 decimal place. Find the upper bound of cost if price is \(\$4\) per kg.\MarkOne \AnswerBox
  \item A time is measured as \(30\) s to the nearest second. State the maximum absolute error.\MarkOne \AnswerBox
\end{enumerate}
