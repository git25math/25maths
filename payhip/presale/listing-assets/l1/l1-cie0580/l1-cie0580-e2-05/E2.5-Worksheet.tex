\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Solve \(3x-11=16\).\MarkOne \AnswerBox
  \item Solve \(5-2x=17\).\MarkOne \AnswerBox
  \item Solve \(\frac{x+4}{3}=7\).\MarkOne \AnswerBox
  \item Solve \(4(2y-1)=3y+20\).\MarkOne \AnswerBox
  \item Solve \(2x+3=5(x-1)\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Quadratic Equation Methods}{Factorising and solving quadratic forms}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Solve \(x^2-9=0\).\MarkOne \AnswerBox
  \item Solve \(x^2+7x+10=0\).\MarkOne \AnswerBox
  \item Solve \(2x^2-5x-3=0\).\MarkOne \AnswerBox
  \item Solve \(x^2=12\).\MarkOne \AnswerBox
  \item Solve \((x-2)(x+5)=0\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Equation Applications}{Simultaneous equations, substitution, and formula solving}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Solve simultaneously: \(x+y=11\), \(x-y=3\).\MarkOne \AnswerBox
  \item Solve simultaneously: \(y=2x+1\), \(x+y=13\).\MarkOne \AnswerBox
  \item Use \(A=\pi r^2\) with \(\pi=\frac{22}{7}\). Find \(r\) when \(A=154\).\MarkOne \AnswerBox
  \item Use \(v=u+at\). Find \(t\) when \(v=29\), \(u=5\), \(a=3\).\MarkOne \AnswerBox
  \item A rectangle has length \(x+3\), width \(x-1\), and area \(60\). Find \(x\).\MarkOne \AnswerBox
\end{enumerate}
