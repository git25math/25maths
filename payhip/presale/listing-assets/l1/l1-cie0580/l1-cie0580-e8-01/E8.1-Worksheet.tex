\SectionHeader{Section A: Probability Foundations}{Sample spaces, simple events, and complements}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A fair six-sided die is rolled. How many outcomes are in the sample space?\MarkOne \AnswerBox
  \item Find the probability of rolling an odd number on a fair six-sided die.\MarkOne \AnswerBox
  \item Find the probability of rolling a prime number on a fair six-sided die.\MarkOne \AnswerBox
  \item A spinner has numbers 1 to 12 equally likely. Find \(P(\text{not a multiple of }3)\).\MarkOne \AnswerBox
  \item A bag has 3 red and 5 blue counters. One counter is chosen at random. Find \(P(\text{red})\).\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Probability Scale Applications}{Interpreting chance in fractions and percentages}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item If \(P(A)=0.37\), find \(P(A')\).\MarkOne \AnswerBox
  \item If \(P(B)=\frac{7}{10}\), find \(P(B')\).\MarkOne \AnswerBox
  \item Two fair coins are tossed. Find probability of getting two heads.\MarkOne \AnswerBox
  \item Two fair coins are tossed. Find probability of at least one head.\MarkOne \AnswerBox
  \item Two fair dice are rolled. Find probability that the sum is 7.\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Intro Probability Reasoning}{Odds and multi-step simple-event contexts}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item If probability of rain is 0.2, write the odds in favor of rain.\MarkOne \AnswerBox
  \item If odds against an event are \(3:2\), find its probability.\MarkOne \AnswerBox
  \item A bag has 2 red, 3 green, and 5 blue counters. One is chosen at random. Find \(P(\text{not blue})\).\MarkOne \AnswerBox
  \item An integer is chosen at random from \(-2,-1,0,1,2,3,4,5\). Find probability it is positive.\MarkOne \AnswerBox
  \item Two fair dice are rolled. Find probability of at least one 6.\MarkOne \AnswerBox
\end{enumerate}
