\SectionHeader{Section A: Construction Principles}{Compass and straightedge foundations}{8--10 mins}
\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item What key property defines the perpendicular bisector of segment \(AB\)?\MarkOne \AnswerBox
  \item What does an angle bisector do to an angle?\MarkOne \AnswerBox
  \item A triangle is constructed with side lengths 5 cm, 6 cm, and 7 cm. Which congruence condition is used?\MarkOne \AnswerBox
  \item With compass centre \(O\) and radius 4 cm, what figure is traced?\MarkOne \AnswerBox
  \item A perpendicular is constructed from point \(P\) to line \(l\). What angle is formed at the foot?\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section B: Standard Construction Outcomes}{Bisectors, perpendiculars, parallels, and loci}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item If \(AB=8\) cm and midpoint \(M\) is constructed, find \(AM\) and \(MB\).\MarkOne \AnswerBox
  \item Using equilateral-triangle construction at point \(A\), what angle size can be constructed exactly?\MarkOne \AnswerBox
  \item A line through point \(P\) is constructed to be parallel to line \(l\). What angle relationship is copied?\MarkOne \AnswerBox
  \item What is the locus of points 3 cm from point \(A\)?\MarkOne \AnswerBox
  \item What is the locus of points equidistant from the x-axis and y-axis?\MarkOne \AnswerBox
\end{enumerate}

\SectionHeader{Section C: Extended Construction Reasoning}{Property proof from construction results}{8--10 mins}
\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item If a line from a circle centre is perpendicular to a chord, what does it do to the chord?\MarkOne \AnswerBox
  \item If \(\angle XYZ=74^\circ\) and we construct its bisector, what are the two resulting angles?\MarkOne \AnswerBox
  \item Constructing a triangle from two sides and the included angle uses which condition name?\MarkOne \AnswerBox
  \item At point \(T\), a tangent is constructed to a circle with centre \(O\). What is the angle between \(OT\) and the tangent?\MarkOne \AnswerBox
  \item What is the locus of points 2 cm from a given straight line \(l\)?\MarkOne \AnswerBox
\end{enumerate}
