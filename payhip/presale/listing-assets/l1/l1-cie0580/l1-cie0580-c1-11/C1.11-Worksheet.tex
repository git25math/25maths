\SectionHeader{Section A: Ratio Basics}{Simplify and compare ratios in different forms}{8--10 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Simplify the ratio \(18:24\).\MarkOne \AnswerBox
  \item Simplify the ratio \(35:49\).\MarkOne \AnswerBox
  \item Simplify the ratio \(0.6:1.5\).\MarkOne \AnswerBox
  \item Simplify the ratio \(250\text{ g}:1\text{ kg}\).\MarkOne \AnswerBox
  \item Simplify the ratio \(45\text{ min}:1.5\text{ h}\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Sharing and Proportion}{Share amounts and use direct proportion}{6--8 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Divide \(\$84\) in the ratio \(2:5\).\MarkOne \AnswerBox
  \item Divide \(96\) sweets in the ratio \(5:3\).\MarkOne \AnswerBox
  \item If \(a:b=4:7\) and \(b=21\), find \(a\).\MarkOne \AnswerBox
  \item On a map with scale \(1:50000\), a road is \(3\) cm long. Find the actual length in km.\MarkOne \AnswerBox
  \item If \(8\) pens cost \(\$6\), find the cost of \(12\) pens.\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Real-life Ratio Problems}{Recipes, unit rates, and best buys}{7--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item In a recipe, flour:sugar \(=3:2\). If flour is \(450\) g, find the sugar needed.\MarkOne \AnswerBox
  \item \(5\) kg of apples cost \(\$8\). Find the cost of \(8\) kg.\MarkOne \AnswerBox
  \item Pack A: \(6\) bars for \(\$4.50\). Pack B: \(8\) bars for \(\$5.60\). Which is the better buy?\MarkOne \AnswerBox
  \item A drink is mixed in the ratio syrup:water \(=1:4\). If water is \(750\) ml, find the syrup amount.\MarkOne \AnswerBox
  \item A \(2.5\) kg bag of rice costs \(\$7.50\). Find the cost per kg.\MarkOne \AnswerBox
\end{enumerate}
