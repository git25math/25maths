% ---- SECTION A
\SectionHeader{Section A: Totals \& Change}{Add/subtract money, give change, multiples}{8--10 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A notebook costs \$2.35 and a pen costs \$1.40. Find the total cost.\MarkOne \AnswerBox
  \item You pay \$10.00 for an item that costs \$6.75. How much change do you receive?\MarkOne \AnswerBox
  \item A drink costs \$1.80. You buy 4 drinks. What is the total cost?\MarkOne \AnswerBox
  \item A bus ticket costs \$2.60. You have \$20.00. How many tickets can you buy \textbf{at most}?\MarkOne \AnswerBox
  \item A sandwich costs \$3.25 and a juice costs \$1.95. You buy one of each. How much change from \$10.00?\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Unit Rates \& Comparisons}{Cost per unit, decimals, best value}{6--8 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item In a shop, 1 kg of apples costs \$3.60. How much do 2.5 kg cost?\MarkOne \AnswerBox
  \item A pack of 6 eggs costs \$2.70. Find the cost of \textbf{one} egg.\MarkOne \AnswerBox
  \item \textbf{Compare prices:}\par\vspace{2pt}
        \hspace{1.5em}\textbullet\ Shop A: 500\,g of rice for \$1.80\par
        \hspace{1.5em}\textbullet\ Shop B: 1.2\,kg of rice for \$4.20\par\vspace{4pt}
        Which shop is cheaper \textbf{per kg}?\MarkOne \AnswerBox
  \item A family budget is \$45.00 for groceries. They spend \$12.60 on fruit, \$18.40 on vegetables, and \$9.75 on bread. How much money is left?\MarkOne \AnswerBox
  \item A taxi charges a fixed fee of \$3.50 plus \$1.80 per km. Find the cost of a 7 km trip.\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Discounts \& Real-life Money}{Savings, exchange rates, multi-item totals}{7--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A T-shirt is discounted from \$18.00 to \$15.30. How much money is saved?\MarkOne \AnswerBox
  \item A movie ticket costs \$9.50. A student pays with a \$20 note. The cashier gives back only \$8.00. How much was the student short-changed?\MarkOne \AnswerBox
  \item You exchange money: 1 USD = 1.35 SGD. You exchange \$40 USD. How many SGD do you get?\MarkOne \AnswerBox
  \item A phone case costs \$7.90. A screen protector costs \$4.60. You buy 2 phone cases and 1 screen protector. What is the total cost?\MarkOne \AnswerBox
  \item A shop offers: \textbf{``Buy 2, get 20\% off the total.''} Each item costs \$12.50. Find the amount you pay for 2 items.\MarkOne \AnswerBox
\end{enumerate}
