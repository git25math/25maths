% ---- SECTION A
\SectionHeader{Section A: Units and Rates}{Convert units and compute rates}{7--9 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Convert 3.6 km to metres.\MarkOne \AnswerBox
  \item Convert 4500 ml to litres.\MarkOne \AnswerBox
  \item A car travels 150 km in 2.5 h. Find average speed in km/h.\MarkOne \AnswerBox
  \item Find density if mass is 540 g and volume is 200 cm\(^3\).\MarkOne \AnswerBox
  \item Pack A: 6 pens for \(\pounds4.50\). Pack B: 10 pens for \(\pounds7.20\). Which is cheaper per pen?\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Number in Context}{Finance and scaling calculations}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A bill is \(\pounds84\) plus 15\% service charge. Find total bill.\MarkOne \AnswerBox
  \item Exchange rate is \(\pounds1=\$1.25\). Convert \(\pounds240\) to dollars.\MarkOne \AnswerBox
  \item A car uses 52 L fuel for 650 km. Find litres per 100 km.\MarkOne \AnswerBox
  \item Income is \(\pounds1250\), expenses are \(\pounds980\). Find monthly saving.\MarkOne \AnswerBox
  \item A recipe for 8 people uses 600 g flour. How much flour is needed for 14 people?\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Multi-step Modelling}{Estimate and solve realistic numeric problems}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Estimate \(19.8\times4.92\) by rounding each factor to 1 significant figure.\MarkOne \AnswerBox
  \item Calculate the exact value of \(19.8\times4.92\).\MarkOne \AnswerBox
  \item A taxi fare is \(\pounds3.50\) plus \(\pounds1.20\) per km. Find fare for 8 km.\MarkOne \AnswerBox
  \item Electricity use is 360 kWh at \(\pounds0.28\) per kWh. Find total cost.\MarkOne \AnswerBox
  \item Monthly salary is \(\pounds2400\) with an 8\% pension deduction. Find take-home pay.\MarkOne \AnswerBox
\end{enumerate}
