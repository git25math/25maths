% ---- SECTION A
\SectionHeader{Section A: Core Probability Facts}{Simple and two-step event probabilities}{7--9 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item On a fair die, find \(P(\text{prime})\).\MarkOne \AnswerBox
  \item A bag has 5 red, 3 blue, 2 green counters. Find \(P(\text{not red})\).\MarkOne \AnswerBox
  \item Two fair coins are tossed. Find \(P(\text{exactly one head})\).\MarkOne \AnswerBox
  \item If independent events have \(P(A)=0.4\), \(P(B)=0.7\), find \(P(A\cap B)\).\MarkOne \AnswerBox
  \item If mutually exclusive events have \(P(A)=0.35\), \(P(B)=0.2\), find \(P(A\cup B)\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Event Relationships}{Use union/intersection/conditional ideas}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item If \(P(A)=0.18\), find \(P(A')\).\MarkOne \AnswerBox
  \item In a class of 40, 24 study French \(F\), 18 study Spanish \(S\), 10 study both. Find \(P(F\cup S)\).\MarkOne \AnswerBox
  \item Using Q7 data, find \(P(F\mid S)\).\MarkOne \AnswerBox
  \item A bag has 4 red and 6 blue. Two draws without replacement. Find \(P(\text{both red})\).\MarkOne \AnswerBox
  \item If success probability is 0.27 in 500 trials, find expected successes.\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Advanced Applications}{Tree and context-driven probability models}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item In two independent trials with \(P(A)=0.3\), find probability of exactly one \(A\).\MarkOne \AnswerBox
  \item If odds in favor of an event are \(3:5\), find the probability.\MarkOne \AnswerBox
  \item If odds against an event are \(4:1\), find the probability of the event.\MarkOne \AnswerBox
  \item True or False: For independent events, \(P(A\mid B)=P(A)\).\MarkOne \AnswerBox
  \item Roll two fair dice. Find \(P(\text{sum}\geq10)\).\MarkOne \AnswerBox
\end{enumerate}
