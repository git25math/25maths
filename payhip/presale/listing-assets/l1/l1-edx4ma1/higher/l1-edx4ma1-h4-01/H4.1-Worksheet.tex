% ---- SECTION A
\SectionHeader{Section A: Core Angle Facts}{Apply foundational angle rules}{7--9 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item In parallel lines, if one corresponding angle is \(68^\circ\), find the alternate angle.\MarkOne \AnswerBox
  \item One angle on a straight line is \(121^\circ\). Find the adjacent angle.\MarkOne \AnswerBox
  \item Angles around a point are \(145^\circ\), \(95^\circ\), and \(x\). Find \(x\).\MarkOne \AnswerBox
  \item Two angles in a triangle are \(48^\circ\) and \(67^\circ\). Find the third angle.\MarkOne \AnswerBox
  \item An exterior angle of a triangle is \(132^\circ\), and one remote interior angle is \(57^\circ\). Find the other remote interior angle.\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Structured Angle Solving}{Use linked angle equations}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item In an isosceles triangle, the vertex angle is \(40^\circ\). Find each base angle.\MarkOne \AnswerBox
  \item In a right triangle, one acute angle is \(34^\circ\). Find the other acute angle.\MarkOne \AnswerBox
  \item In triangle \(ABC\), \(AB=AC\), \(\angle B=x+10\), \(\angle C=2x-5\). Find \(x\).\MarkOne \AnswerBox
  \item Co-interior angles between parallel lines are \(3x+12\) and \(2x+33\). Find \(x\).\MarkOne \AnswerBox
  \item Angles in a quadrilateral are \(95^\circ\), \(83^\circ\), \(102^\circ\), and \(x\). Find \(x\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Geometric Reasoning}{Solve multi-step angle problems}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item An exterior angle of a triangle is \(146^\circ\), and one opposite interior angle is \(68^\circ\). Find the other opposite interior angle.\MarkOne \AnswerBox
  \item In an isosceles triangle, base angles are \(2y+6\) and \(3y-9\). Find \(y\).\MarkOne \AnswerBox
  \item In a right triangle, hypotenuse is 13 and one leg is 5. Find the other leg.\MarkOne \AnswerBox
  \item Vertically opposite angles are \(4x-7\) and \(3x+18\). Find \(x\).\MarkOne \AnswerBox
  \item True or False: The sum of interior angles of a pentagon is \(540^\circ\).\MarkOne \AnswerBox
\end{enumerate}
