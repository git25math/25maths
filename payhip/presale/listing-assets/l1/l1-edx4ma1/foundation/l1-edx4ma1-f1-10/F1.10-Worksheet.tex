% ---- SECTION A
\SectionHeader{Section A: Everyday Number Skills}{Money, fractions, unit conversion}{7--9 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Find the total cost of \(3\) cinema tickets at \(\pounds 7.50\) each.\MarkOne \AnswerBox
  \item Find the change from \(\pounds 20\) after buying an item costing \(\pounds 13.65\).\MarkOne \AnswerBox
  \item Find \(\frac{3}{5}\) of \(45\).\MarkOne \AnswerBox
  \item Find \(0.25\) of \(84\).\MarkOne \AnswerBox
  \item Convert \(2.4\,\text{km}\) to metres.\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Multi-Step Problems}{Rates, percentages, and scaling}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Twelve eggs cost \(\pounds 2.40\). Find the cost of \(5\) eggs.\MarkOne \AnswerBox
  \item Find \(15\%\) tip on a meal costing \(\pounds 32\).\MarkOne \AnswerBox
  \item A phone battery is at \(80\%\), then drops by \(35\) percentage points. What is the new battery percentage?\MarkOne \AnswerBox
  \item A train travels \(135\,\text{km}\) in \(2\) h \(15\) min. Find the average speed in km/h.\MarkOne \AnswerBox
  \item A recipe uses \(250\,\text{g}\) flour for \(10\) muffins. How much flour is needed for \(14\) muffins?\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Decision and Interpretation}{Best-buy and practical reasoning}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Which is better value: \(750\,\text{ml}\) for \(\pounds 1.20\) or \(1.5\,\text{L}\) for \(\pounds 2.10\)?\MarkOne \AnswerBox
  \item You have a budget of \(\pounds 50\). You buy \(3\) notebooks at \(\pounds 4.20\) each and \(2\) pens at \(\pounds 1.35\) each. How much money is left?\MarkOne \AnswerBox
  \item Electricity costs \(28\)p per kWh. Find the cost of \(12.5\) kWh.\MarkOne \AnswerBox
  \item An item costs \(\pounds 48\), then gets \(25\%\) off. Delivery is \(\pounds 4\). Find the final amount paid.\MarkOne \AnswerBox
  \item A map scale shows \(1\,\text{cm}\) represents \(2.5\,\text{km}\). Find the real distance for \(7.2\,\text{cm}\).\MarkOne \AnswerBox
\end{enumerate}
