% ---- SECTION A
\SectionHeader{Section A: Fraction Basics}{Equivalent forms, simplification, conversion}{7--9 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Simplify \(\frac{12}{18}\).\MarkOne \AnswerBox
  \item Convert \(2\frac{1}{3}\) to an improper fraction.\MarkOne \AnswerBox
  \item Convert \(\frac{17}{5}\) to a mixed number.\MarkOne \AnswerBox
  \item Which fraction is greater: \(\frac{3}{8}\) or \(\frac{5}{12}\)?\MarkOne \AnswerBox
  \item Write a fraction equivalent to \(\frac{3}{4}\) with denominator \(20\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Fraction Operations}{Addition, subtraction, multiplication, division}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Evaluate \(\frac{2}{5}+\frac{1}{5}\).\MarkOne \AnswerBox
  \item Evaluate \(\frac{7}{8}-\frac{1}{4}\).\MarkOne \AnswerBox
  \item Evaluate \(\frac{3}{7}\times\frac{14}{15}\).\MarkOne \AnswerBox
  \item Evaluate \(\frac{5}{6}\div\frac{10}{9}\).\MarkOne \AnswerBox
  \item Evaluate \(2\frac{1}{2}+1\frac{3}{4}\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Problem Solving}{Fractions in context and simple algebra}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A recipe needs \(\frac{3}{4}\) cup of sugar per batch. How much sugar is needed for \(2\) batches?\MarkOne \AnswerBox
  \item A student reads \(\frac{2}{5}\) of a book on Monday and \(\frac{1}{4}\) on Tuesday. What fraction of the book has been read?\MarkOne \AnswerBox
  \item What fraction of the same book is left unread?\MarkOne \AnswerBox
  \item Evaluate \(\frac{x}{3}\) when \(x=\frac{5}{2}\).\MarkOne \AnswerBox
  \item A tank is \(\frac{3}{4}\) full. If \(\frac{1}{6}\) of the full tank is used, what fraction of the tank remains?\MarkOne \AnswerBox
\end{enumerate}
