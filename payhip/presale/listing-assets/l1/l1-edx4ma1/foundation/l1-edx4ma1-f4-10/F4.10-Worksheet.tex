% ---- SECTION A
\SectionHeader{Section A: Core Volumes}{Cube, cuboid, prism, cylinder}{7--9 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Find the volume of a cube with side 4 cm.\MarkOne \AnswerBox
  \item Find the total surface area of the same cube.\MarkOne \AnswerBox
  \item Find the volume of a cuboid with dimensions \(8\times5\times3\) cm.\MarkOne \AnswerBox
  \item Find the total surface area of that cuboid.\MarkOne \AnswerBox
  \item Find the volume of a cylinder with radius 3 cm and height 10 cm in terms of \(\pi\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Surface Areas}{Cube/cuboid and related calculations}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A prism has cross-sectional area \(12\text{ cm}^2\) and length 7 cm. Find its volume.\MarkOne \AnswerBox
  \item A triangular prism has triangular cross-sectional area \(15\text{ cm}^2\) and length 4 cm. Find volume.\MarkOne \AnswerBox
  \item Find the volume of a cone with radius 3 cm and height 12 cm in terms of \(\pi\).\MarkOne \AnswerBox
  \item Find the volume of a cylinder of diameter 10 cm and height 6 cm in terms of \(\pi\).\MarkOne \AnswerBox
  \item If the volume of a cube is \(343\text{ cm}^3\), find its side length.\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Mixed 3D Applications}{Capacity and multi-step geometry}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item A cuboid has volume \(180\text{ cm}^3\) and base area \(30\text{ cm}^2\). Find its height.\MarkOne \AnswerBox
  \item Convert \(2.5\text{ m}^3\) to litres.\MarkOne \AnswerBox
  \item Find the area of a circular base with radius 4 cm in terms of \(\pi\).\MarkOne \AnswerBox
  \item How many edges does a cube have?\MarkOne \AnswerBox
  \item How many faces does a triangular prism have?\MarkOne \AnswerBox
\end{enumerate}
