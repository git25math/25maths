% ---- SECTION A
\SectionHeader{Section A: Conversion Skills}{Convert between ordinary numbers and standard form}{7--9 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Write \(56{,}000\) in standard form.\MarkOne \AnswerBox
  \item Write \(0.00072\) in standard form.\MarkOne \AnswerBox
  \item Write \(3.4\times 10^5\) as an ordinary number.\MarkOne \AnswerBox
  \item Write \(9.1\times 10^{-3}\) as an ordinary number.\MarkOne \AnswerBox
  \item Which is greater: \(6\times10^3\) or \(5.9\times10^3\)?\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Standard Form Operations}{Add/subtract (aligned powers), multiply, divide}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Evaluate \((2\times10^4)+(3\times10^4)\). Give your answer in standard form.\MarkOne \AnswerBox
  \item Evaluate \((7\times10^6)-(2\times10^6)\). Give your answer in standard form.\MarkOne \AnswerBox
  \item Evaluate \((3\times10^2)(4\times10^3)\).\MarkOne \AnswerBox
  \item Evaluate \((8\times10^5)\div(2\times10^2)\).\MarkOne \AnswerBox
  \item Evaluate \((6\times10^7)(5\times10^{-3})\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Applications}{Context problems involving very large/small values}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Light travels at \(3\times10^8\,\text{m/s}\). How far does it travel in \(0.02\,\text{s}\)?\MarkOne \AnswerBox
  \item One bacterium has mass \(4\times10^{-12}\,\text{kg}\). Find the mass of \(5\times10^6\) bacteria.\MarkOne \AnswerBox
  \item The Earth--Sun distance is \(1.5\times10^8\,\text{km}\). Write this as an ordinary number.\MarkOne \AnswerBox
  \item Write \(0.0000045\) in standard form.\MarkOne \AnswerBox
  \item Evaluate \((9\times10^4)\div(3\times10^{-2})\).\MarkOne \AnswerBox
\end{enumerate}
