% ---- SECTION A
\SectionHeader{Section A: Ratio Foundations}{Equivalent ratios and sharing}{7--9 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Complete the ratio: \(3:4 = 12:\square\).\MarkOne \AnswerBox
  \item Simplify \(18:24\).\MarkOne \AnswerBox
  \item Share \(\pounds35\) in the ratio \(2:5\). Find the larger share.\MarkOne \AnswerBox
  \item If \(A:B=5:8\) and \(B=24\), find \(A\).\MarkOne \AnswerBox
  \item If \(m:n=7:3\) and \(m+n=50\), find \(n\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Direct Proportion}{Find constants and scale values}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item \(y\) is directly proportional to \(x\). If \(y=15\) when \(x=3\), find \(y\) when \(x=7\).\MarkOne \AnswerBox
  \item \(y\) is directly proportional to \(x\). If \(y=8\) when \(x=2\), write a formula for \(y\) in terms of \(x\).\MarkOne \AnswerBox
  \item 12 litres of fuel cost \(\pounds18\). Find the cost of 20 litres.\MarkOne \AnswerBox
  \item 250 g of flour makes 10 biscuits. How much flour is needed for 14 biscuits?\MarkOne \AnswerBox
  \item A map scale is \(1:50{,}000\). If two points are 3 cm apart on the map, how far apart are they in km?\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Inverse and Context}{Inverse proportion and real contexts}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item \(p\) is inversely proportional to \(q\). If \(p=10\) when \(q=4\), find \(p\) when \(q=5\).\MarkOne \AnswerBox
  \item 6 workers complete a job in 15 days. Assuming inverse proportion, how many days for 10 workers?\MarkOne \AnswerBox
  \item Distance is directly proportional to time at constant speed. If a car travels 60 km in 1.5 h, find its speed.\MarkOne \AnswerBox
  \item If \(x\) doubles in a direct proportion model \(y\propto x\), what happens to \(y\)?\MarkOne \AnswerBox
  \item A paint mix has red:blue = \(4:1\). If total volume is 45 L, find the blue paint volume.\MarkOne \AnswerBox
\end{enumerate}

  \faLightbulb\ \textbf{Teacher Tip:}
  Have students state whether the relationship is direct or inverse before doing any calculation.
