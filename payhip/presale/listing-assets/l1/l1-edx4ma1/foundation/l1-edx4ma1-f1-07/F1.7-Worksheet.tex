% ---- SECTION A
\SectionHeader{Section A: Ratio Basics}{Simplifying and sharing using ratios}{7--9 mins}

\begin{enumerate}[label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Simplify the ratio \(18:24\).\MarkOne \AnswerBox
  \item Share \(\pounds 35\) in the ratio \(2:5\). Find the larger share.\MarkOne \AnswerBox
  \item Write \(250\,\text{ml}:1\,\text{L}\) as a ratio in simplest whole-number form.\MarkOne \AnswerBox
  \item If \(a:b=3:7\) and \(a=12\), find \(b\).\MarkOne \AnswerBox
  \item Divide \(48\) in the ratio \(1:3\). Find the smaller part.\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION B
\SectionHeader{Section B: Proportion Methods}{Direct and inverse proportion calculations}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item Five pencils cost \(\pounds 3.50\). Find the cost of \(8\) pencils.\MarkOne \AnswerBox
  \item Four workers complete a job in \(15\) days. At the same rate, how many days do \(6\) workers need?\MarkOne \AnswerBox
  \item A recipe for \(3\) people uses \(240\,\text{g}\) of rice. How much is needed for \(5\) people?\MarkOne \AnswerBox
  \item A map scale is \(1:50{,}000\). A map distance is \(3\,\text{cm}\). Find the real distance in km.\MarkOne \AnswerBox
  \item \(y\) is directly proportional to \(x\). If \(y=18\) when \(x=6\), find \(y\) when \(x=10\).\MarkOne \AnswerBox
\end{enumerate}

% ---- SECTION C
\SectionHeader{Section C: Real Contexts}{Scale, rates, and reasoning with proportion}{8--10 mins}

\begin{enumerate}[resume, label=\qnum{\arabic*}, leftmargin=3.2em, itemsep=12pt]
  \item \(y\) is inversely proportional to \(x\). If \(y=12\) when \(x=4\), find \(y\) when \(x=6\).\MarkOne \AnswerBox
  \item A red:blue paint mix ratio is \(3:2\). If total paint is \(25\,\text{L}\), find the amount of red paint.\MarkOne \AnswerBox
  \item In a school, boys:girls \(=4:5\). If there are \(270\) students in total, find the number of girls.\MarkOne \AnswerBox
  \item True or False: \(6:8\) is equivalent to \(9:12\).\MarkOne \AnswerBox
  \item At constant speed, a car travels \(120\,\text{km}\) in \(2\) hours. How long for \(210\,\text{km}\)?\MarkOne \AnswerBox
\end{enumerate}
